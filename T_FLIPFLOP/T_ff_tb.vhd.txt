library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity t_ff_tb is
end t_ff_tb;

architecture Behavioral of t_ff_tb is

    component t_ff
        Port (
            T   : in  STD_LOGIC;
            CLK : in  STD_LOGIC;
            Q   : out STD_LOGIC
        );
    end component;

    signal T   : STD_LOGIC := '0';
    signal CLK : STD_LOGIC := '0';
    signal Q   : STD_LOGIC;

begin

    uut: t_ff
        port map (
            T   => T,
            CLK => CLK,
            Q   => Q
        );

    -- Clock Generation
    clk_process : process
    begin
        while true loop
            CLK <= '0';
            wait for 5 ns;
            CLK <= '1';
            wait for 5 ns;
        end loop;
    end process;

    -- Stimulus
    stim_proc : process
    begin
        T <= '0';
        wait for 20 ns;

        T <= '1';
        wait for 40 ns;

        T <= '0';
        wait for 20 ns;

        T <= '1';
        wait for 40 ns;

        wait;
    end process;

end Behavioral;
