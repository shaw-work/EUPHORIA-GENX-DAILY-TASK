library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity t_ff is
    Port (
        T   : in  STD_LOGIC;
        CLK : in  STD_LOGIC;
        Q   : out STD_LOGIC
    );
end t_ff;

architecture Behavioral of t_ff is
    signal q_reg : STD_LOGIC := '0';
begin

    process(CLK)
    begin
        if rising_edge(CLK) then
            if T = '1' then
                q_reg <= not q_reg;
            end if;
        end if;
    end process;

    Q <= q_reg;

end Behavioral;
